import pandas as pd

# Sample sales data
data = {
    "Product": ["Laptop", "Phone", "Chair", "Table"],
    "Sales": [50000, 30000, 10000, 15000]
}

# Create DataFrame
df = pd.DataFrame(data)

# Calculate total revenue
total_revenue = df["Sales"].sum()

print(df)
